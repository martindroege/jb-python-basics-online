User-Input Skript¶

# Eingabe Name
name = input("Wie heißt du? >>>")

# Eingabe Alter
alter = input("Wie alt bist du? >>>")

# Eingabe Wohnort
ort = input("Wo wohnst du? >>>")

jahr = 2021 - int(alter)

# Ausgabe
print(f"""
\nHallo {name}, schön, dass du da bist!\n
Du bist {jahr} geboren.\n
{ort} ist der beste Ort auf dem Planeten.""")
